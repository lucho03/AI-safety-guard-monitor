#1 Install WSL (for Windows)
wsl --install
sudo apt update && sudo apt upgrade

#2 Install mosquitto MQTT broker
sudo apt update && sudo apt install mosquitto

#3 Create Python Virtual Environment
python -m venv Venv

#4 Create certs folder
rm -rf certs
mkdir certs
cd certs

#5 Create CA
openssl genrsa -out ca.key 2048
openssl req -x509 -new -nodes -key ca.key -sha256 -days 3650 -out ca.crt

#6 Create server/client key + CSR
openssl genrsa -out server.key 2048
openssl req -new -key server.key -out server.csr -subj "/CN=localhost"

#7 Sign via CA
openssl x509 -req \
-in server.csr \
-CA ca.crt \
-CAkey ca.key \
-CAcreateserial \
-out server.crt \
-days 365 -sha256

#8 Check
openssl x509 -in server.crt -text -noout | grep Issuer

#9 Create users
sudo mosquitto_passwd {path}/passwd {user}
